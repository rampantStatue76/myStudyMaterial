    ```vue
    // 父组件
    <template>
        <Child v-model:ww = 'w'/>
        <button @click="changeW">ww加一</button>
    </template>
    <script setup>
        const w = ref(0)
        const changeW = () => {
            w.value ++
        }
    </script>

    //子组件
    <template>
        <span>{{ wValue }}</span>
    </template>
    <script setup>
        import { defineModel } from 'vue'

        const wValue = defineModel('ww')
    </script>
    ```
